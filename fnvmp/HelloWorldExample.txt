#include "nvse/PluginAPI.h"
#include "nvse/GameAPI.h"
#include "nvse/GameObjects.h"

// --------------------------------------------------
// Globals
// --------------------------------------------------

IDebugLog gLog("luka_test\\hello_world_mvp.log");

PluginHandle g_pluginHandle = kPluginHandle_Invalid;
NVSEMessagingInterface* g_messaging = nullptr;

// Accumulated time in seconds
static float g_timeAccumulator = 0.0f;

// --------------------------------------------------
// Tick function
// --------------------------------------------------

void Tick_OncePerSecond()
{
	_MESSAGE("hello world");
}

// --------------------------------------------------
// Message handler
// --------------------------------------------------

// Main loop timing (monotonic)
static double g_accum = 0.0;
static LARGE_INTEGER g_freq{};
static LARGE_INTEGER g_last{};

void MessageHandler(NVSEMessagingInterface::Message* msg)
{
	// log all message types
	//_MESSAGE("Message received: type %u from %s", msg->type, msg->sender);

	if (msg->type != NVSEMessagingInterface::kMessage_MainGameLoop)
		return;

	if (g_freq.QuadPart == 0)
	{
		QueryPerformanceFrequency(&g_freq);
		QueryPerformanceCounter(&g_last);
		return;
	}

	LARGE_INTEGER now;
	QueryPerformanceCounter(&now);

	double dt = double(now.QuadPart - g_last.QuadPart) / double(g_freq.QuadPart);
	g_last = now;

	g_accum += dt;
	if (g_accum >= 1.0)
	{
		g_accum -= 1.0;
		Tick_OncePerSecond();
	}
}

// --------------------------------------------------
// NVSE entry points
// --------------------------------------------------

extern "C"
{

	bool NVSEPlugin_Query(const NVSEInterface* nvse, PluginInfo* info)
	{
		info->infoVersion = PluginInfo::kInfoVersion;
		info->name = "HelloWorldMVP";
		info->version = 1;

		if (nvse->nvseVersion < PACKED_NVSE_VERSION)
			return false;

		if (!nvse->isEditor && nvse->runtimeVersion < RUNTIME_VERSION_1_4_0_525)
			return false;

		return true;
	}

	bool NVSEPlugin_Load(NVSEInterface* nvse)
	{
		g_pluginHandle = nvse->GetPluginHandle();

		g_messaging = static_cast<NVSEMessagingInterface*>(
			nvse->QueryInterface(kInterface_Messaging)
			);

		if (!g_messaging)
			return false;

		g_messaging->RegisterListener(
			g_pluginHandle,
			"NVSE",
			MessageHandler
		);

		_MESSAGE("HelloWorldMVP loaded");
		return true;
	}

} // extern "C"
